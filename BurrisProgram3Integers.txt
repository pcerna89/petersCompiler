CLASS BurrisProgram3Integers{
    VAR INT1, INT2, INT3, LARGEST;
    CIN INT1;
    CIN INT2;
    CIN INT3;
    IF INT1 > INT2 THEN {
        IF INT1 > INT3 THEN {
            LARGEST = INT1;
        }
        ELSE {
            LARGEST = INT3; 
        }
    }
    ELSE {
        IF INT2 > INT3 THEN {
            LARGEST = INT2; 
        }
        ELSE {
            LARGEST = INT3;
        }
    }
    COUT LARGEST;
}
